Defining a function with time series¶

In [32]:
def plot_timeseies (axes,x,y,color,xlabel,ylabel):
    axes.plot(x,y,color=color)
    axes.set_xlabel(xlabel)
    axes.set_ylabel(ylabel,color=color)
    axes.tick_params("y",colors=color)
plt.show()
